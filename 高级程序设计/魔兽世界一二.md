	int number;  //编号
	int hp;  //生命值

		sum_number++;

		number = n;
		hp = hhp;

		number = n;
		arm = n % 3;
		mor = (float)m / (float)hhp;

		cout << " red dragon " << number << " born with strength " << hp << "," << num_sum << " dragon in red headquarter" << endl;

};

class setclock {
private:
	int t;
public:
	setclock() {
		t = 0;
	}
	void clear() {
		t = 0;
	}
	void add() {
		t++;
	}
	void show() {
		if (t < 10) cout << "00" << t;
		if (t >= 10 && t < 100) cout << "0" << t;
		if (t >= 100) cout << t;
	}
	~setclock(){}
};

void casetest(int n) {
	int m;
	cin >> m;
	int m_red = m;
	int m_blue = m;
	int hp[6] = { 0 };
	int min_hp = 100000;
	for (int i = 1; i <= 5; i++) {
		cin >> hp[i];
		min_hp = min(hp[i], min_hp);
	}
	cout << "Case: " << n << endl;
	rn_d = 0, rn_n = 0, rn_i = 0, rn_l = 0, rn_w = 0;
	bn_d = 0, bn_n = 0, bn_i = 0, bn_l = 0, bn_w = 0;
	now_clock.clear();
	int i_red = 1;
	int i_blue = 1;
	int num_red = 0;
	int num_blue = 0;
	int check_red = 1;
	int check_blue = 1;
	while (m_red >= min_hp || m_blue >= min_hp) {
		if (check_red && m_red < min_hp) {
			check_red = 0;
			now_clock.show();
			cout << " red headquarter stops making warriors" << endl;
		}
		if (check_blue && m_blue < min_hp) {
			check_blue = 0;
			now_clock.show();
			cout << " blue headquarter stops making warriors" << endl;
		}
		while (m_red >= min_hp) {
			if (m_red >= hp[order_red[i_red]]) {
				m_red -= hp[order_red[i_red]];
				num_red++;
				now_clock.show();
				createandshow(order_red[i_red], hp, num_red, m_red, 1);
				i_red = i_red % 5 + 1;
				break;
			}
			i_red = i_red % 5 + 1;
		}
		while (m_blue >= min_hp) {
			if (m_blue >= hp[order_blue[i_blue]]) {
				m_blue -= hp[order_blue[i_blue]];
				num_blue++;
				now_clock.show();
				createandshow(order_blue[i_blue], hp, num_blue, m_blue, 2);
				i_blue = i_blue % 5 + 1;
				break;
			}
			i_blue = i_blue % 5 + 1;
		}
		now_clock.add();
	}
	if (check_red && m_red < min_hp) {
		check_red = 0;
		now_clock.show();
		cout << " red headquarter stops making warriors" << endl;
	}
	if (check_blue && m_blue < min_hp) {
		check_blue = 0;
		now_clock.show();
		cout << " blue headquarter stops making warriors" << endl;
	}
}

int main() {
	int nn;
	cin >> nn;
	while (nn) {
		casetest(nn);
		nn--;
	}
}
